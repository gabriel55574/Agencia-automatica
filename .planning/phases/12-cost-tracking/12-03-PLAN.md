    **Step 1: Create `src/components/squad/RunCostBadge.tsx`:**

    ```typescript
    import { Badge } from '@/components/ui/badge'
    import { formatCost, formatTokensCompact } from '@/lib/costs/format'

    interface RunCostBadgeProps {
      tokenCount: number | null
      estimatedCost: number | null
    }

    export function RunCostBadge({ tokenCount, estimatedCost }: RunCostBadgeProps) {
      // Only render if we have at least one value
      if (tokenCount === null && estimatedCost === null) return null

      const costStr = estimatedCost !== null ? formatCost(estimatedCost) : '—'
      const tokenStr = tokenCount !== null ? `${formatTokensCompact(tokenCount)} tokens` : ''

      const label = tokenCount !== null && estimatedCost !== null
        ? `${costStr} | ${tokenStr}`
        : tokenCount !== null
          ? tokenStr
          : costStr

      const ariaLabel = tokenCount !== null
        ? `Cost: ${costStr}, ${tokenCount.toLocaleString('en-US')} tokens`
        : `Cost: ${costStr}`

      return (
        <Badge
          variant="outline"
          className="text-xs font-mono"
          aria-label={ariaLabel}
        >
          {label}
        </Badge>
      )
    }
    ```

    NOT a client component — pure presentational, receives data as props. No hooks or event handlers.

    Format per UI-SPEC COST-01 section:
    - Badge variant="outline" with text-xs font-mono
    - Format: "$X.XX | XX.Xk tokens" (dollar cost pipe-separated from token count)
    - Token count uses compact format (formatTokensCompact)
    - Returns null if both tokenCount and estimatedCost are null
    - aria-label with full token count for accessibility
