    Create `src/lib/costs/format.ts` with all formatting utilities per the UI-SPEC number formatting rules.

    **formatCost(amount: number | null): string**
    - Returns "$0.00" if null/undefined
    - Returns dollar-formatted string: `$${amount.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`
    - Example: formatCost(1234.56) returns "$1,234.56"

    **formatTokensCompact(tokens: number | null): string**
    - Returns "0" if null/undefined
    - Under 1000: return raw number as string (e.g., "850")
    - 1k-999k: return "X.Xk" or "Xk" if no remainder (e.g., "12.3k", "5k")
    - 1M+: return "X.XM" or "XM" if no remainder (e.g., "1.2M", "2M")

    **formatTokensFull(tokens: number | null): string**
    - Returns "0" if null/undefined
    - Returns comma-separated: `tokens.toLocaleString('en-US')` (e.g., "100,000")

    **formatTrend(current: number, previous: number | null): string**
    - Returns dash "—" if previous is null or 0
    - Returns "+X%", "-X%", or "0%" (no decimals, rounded)
    - Positive change = cost went up (red in UI), negative = cost went down (green in UI)

    **formatMonth(month: string): string**
    - Input: "YYYY-MM" (e.g., "2026-04")
    - Output: "MMMM YYYY" (e.g., "April 2026")
    - Implementation: `new Date(year, mon - 1).toLocaleDateString('en-US', { month: 'long', year: 'numeric' })`

    **getCurrentMonth(): string**
    - Returns current date in "YYYY-MM" format
    - Implementation: pad month to 2 digits

    **getLastMonths(count: number): string[]**
    - Returns array of last N months in "YYYY-MM" format, most recent first
    - Used by MonthSelector for dropdown options
  </action>

    **fetchMonthlyCostBreakdown(month: string): Promise&lt;CostBreakdownRow[]&gt;**

    Parameter `month` is in "YYYY-MM" format (e.g., "2026-04").

    1. Calculate date range:
       ```typescript
       const startDate = `${month}-01T00:00:00.000Z`
       const [year, mon] = month.split('-').map(Number)
       const nextMonth = mon === 12
         ? `${year + 1}-01`
         : `${year}-${String(mon + 1).padStart(2, '0')}`
       const endDate = `${nextMonth}-01T00:00:00.000Z`
       ```

    2. Calculate previous month range for trend:
       ```typescript
       const prevMonth = mon === 1
         ? `${year - 1}-12`
         : `${year}-${String(mon - 1).padStart(2, '0')}`
       const prevStartDate = `${prevMonth}-01T00:00:00.000Z`
       ```

    3. Run queries (Supabase JS client does not support GROUP BY, so aggregate in TypeScript):
       ```typescript
       const supabase = await createClient()

       const [currentResult, prevResult] = await Promise.all([
         supabase
           .from('squad_jobs')
           .select('client_id, token_count, estimated_cost_usd')
           .eq('status', 'completed')
           .not('token_count', 'is', null)
           .gte('completed_at', startDate)
           .lt('completed_at', endDate),
         supabase
           .from('squad_jobs')
           .select('client_id, estimated_cost_usd')
           .eq('status', 'completed')
           .not('token_count', 'is', null)
           .gte('completed_at', prevStartDate)
           .lt('completed_at', startDate),
       ])
       ```

    4. Fetch client names for all unique client_ids from current month:
       ```typescript
       const clientIds = [...new Set((currentResult.data ?? []).map(j => j.client_id))]
       const { data: clients } = clientIds.length > 0
         ? await supabase.from('clients').select('id, name').in('id', clientIds)
         : { data: [] }
       const clientNameMap = new Map((clients ?? []).map(c => [c.id, c.name]))
       ```

    5. Aggregate current month by client using Map:
       ```typescript
       const currentByClient = new Map<string, { run_count: number; total_tokens: number; total_cost: number }>()
       for (const job of currentResult.data ?? []) {
         const entry = currentByClient.get(job.client_id) ?? { run_count: 0, total_tokens: 0, total_cost: 0 }
         entry.run_count += 1
         entry.total_tokens += (job.token_count as number) ?? 0
         entry.total_cost += Number(job.estimated_cost_usd ?? 0)
         currentByClient.set(job.client_id, entry)
       }
       ```

    6. Aggregate previous month by client:
       ```typescript
       const prevByClient = new Map<string, number>()
       for (const job of prevResult.data ?? []) {
         prevByClient.set(
           job.client_id,
           (prevByClient.get(job.client_id) ?? 0) + Number(job.estimated_cost_usd ?? 0)
         )
       }
       ```

    7. Build and sort rows:
       ```typescript
       const rows: CostBreakdownRow[] = Array.from(currentByClient.entries()).map(
         ([clientId, data]) => ({
           client_id: clientId,
           client_name: clientNameMap.get(clientId) ?? 'Unknown',
           run_count: data.run_count,
           total_tokens: data.total_tokens,
           total_cost: Math.round(data.total_cost * 100) / 100,
           previous_month_cost: prevByClient.has(clientId)
             ? Math.round((prevByClient.get(clientId) ?? 0) * 100) / 100
             : null,
         })
       )
       rows.sort((a, b) => b.total_cost - a.total_cost)
       return rows
       ```

    **fetchMonthlyCostSummary(month: string): Promise&lt;MonthlyCostSummary&gt;**

    Reuses fetchMonthlyCostBreakdown to avoid query duplication:
    ```typescript
    export async function fetchMonthlyCostSummary(month: string): Promise<MonthlyCostSummary> {
      const rows = await fetchMonthlyCostBreakdown(month)
      const total_cost = Math.round(rows.reduce((sum, r) => sum + r.total_cost, 0) * 100) / 100
      const total_tokens = rows.reduce((sum, r) => sum + r.total_tokens, 0)
      const top_clients = rows.slice(0, 5).map(r => ({
        client_id: r.client_id,
        client_name: r.client_name,
        total_cost: r.total_cost,
      }))
      const prevCosts = rows.filter(r => r.previous_month_cost !== null)
      const previous_month_total = prevCosts.length > 0
        ? Math.round(prevCosts.reduce((sum, r) => sum + (r.previous_month_cost ?? 0), 0) * 100) / 100
        : null
      return { total_cost, total_tokens, top_clients, previous_month_total }
    }
    ```

    import { useRouter, useSearchParams } from 'next/navigation'
    import {
      Select, SelectContent, SelectItem, SelectTrigger, SelectValue,

    import { formatMonth, getLastMonths } from '@/lib/costs/format'

    interface MonthSelectorProps {

    export function MonthSelector({ currentMonth }: MonthSelectorProps) {
      const router = useRouter()

      const months = getLastMonths(12)

      function handleMonthChange(value: string) {
